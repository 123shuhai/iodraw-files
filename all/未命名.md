```mermaid
graph TD
    A[输入账号和密码] --> B{账号是否存在}
    B -->|存在| C{验证账号密码}
    B -->|不存在| D[提示账号不存在，重新输入]
    D --> A
    C -->|正确| E[登录成功]
    C -->|错误| F[提示密码错误，重新输入]
    F --> A    

    digraph G {
    rankdir=LR;
    node [shape=rectangle];
    A [label="输入账号和密码"];
    B [label="账号是否存在" shape=diamond];
    C [label="验证账号密码" shape=diamond];
    D [label="提示账号不存在，重新输入"];
    E [label="登录成功"];
    F [label="提示密码错误，重新输入"];

    A -> B;
    B -> C [label="存在"];
    B -> D [label="不存在"];
    D -> A;
    C -> E [label="正确"];
    C -> F [label="错误"];
    F -> A;
}
```
